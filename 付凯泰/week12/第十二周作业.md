def generate_token(apikey: str, exp_seconds: int = 3600) -> str:
    """
    生成智谱接口 token
    apikey 格式示例：id.secret
    """
    try:
        key_id, secret = apikey.split(".")
    except Exception as e:
        raise ValueError("apikey 格式错误，应为 'id.secret'") from e

    payload = {
        "api_key": key_id,
        "exp": int(round(time.time() * 1000)) + exp_seconds * 1000,
        "timestamp": int(round(time.time() * 1000)),
    }
    token = jwt.encode(
        payload,
        secret,
        algorithm="HS256",
        headers={"alg": "HS256", "sign_type": "SIGN"},
    )
    return token


class GLMClient:

        self.api_key = api_key
        self.model = model
        self.url = "https://open.bigmodel.cn/api/paas/v4/chat/completions"

    def chat(self, messages: List[Dict[str, str]], temperature: float = 0.1) -> str:

class DBParser:
    def __init__(self, db_url: str):
        self.db_url = db_url
        self.engine = create_engine(db_url, echo=False, future=True)
        self.inspector = inspect(self.engine)
        self.table_names = self.inspector.get_table_names()

    def get_schema_text(self) -> str:
        """
        输出整个数据库的 schema 摘要，供大模型参考
        """
        lines = []
        lines.append("数据库中包含以下表：")

        for table_name in self.table_names:
            columns = self.inspector.get_columns(table_name)
            col_desc = ", ".join([f"{c['name']} ({c['type']})" for c in columns])
            lines.append(f"- {table_name}: {col_desc}")

        fk_lines = []
        for table_name in self.table_names:
            foreign_keys = self.inspector.get_foreign_keys(table_name)
            for fk in foreign_keys:
                constrained_columns = ", ".join(fk.get("constrained_columns", []))
                referred_table = fk.get("referred_table", "")
                referred_columns = ", ".join(fk.get("referred_columns", []))
                fk_lines.append(
                    f"- {table_name}.{constrained_columns} -> {referred_table}.{referred_columns}"
                )

        if fk_lines:
            lines.append("\n外键关系：")
            lines.extend(fk_lines)

        return "\n".join(lines)

    def is_safe_sql(self, sql: str) -> bool:
        """
        只允许 SELECT / WITH 查询
        """
        normalized = re.sub(r"\s+", " ", sql.strip().lower())

        if not (normalized.startswith("select") or normalized.startswith("with")):
            return False

        dangerous_keywords = [
            " insert ", " update ", " delete ", " drop ", " alter ",
            " truncate ", " create ", " replace ", " attach ", " detach ",
            " pragma "
        ]
        padded = f" {normalized} "

        for kw in dangerous_keywords:
            if kw in padded:
                return False
        return True

    def execute_sql(self, sql: str) -> Tuple[List[str], List[Dict[str, Any]]]:
        """
        执行 SQL，返回列名和结果列表
        """
        sql = sql.strip().rstrip(";")
        if not self.is_safe_sql(sql):
            raise ValueError(f"检测到非安全 SQL，不予执行：{sql}")

        try:
            with self.engine.connect() as conn:
                result = conn.execute(text(sql))
                rows = result.fetchall()
                columns = list(result.keys())

            data = [dict(zip(columns, row)) for row in rows]
            return columns, data

        except SQLAlchemyError as e:
            raise RuntimeError(f"SQL 执行失败：{e}") from e

QUESTION_REWRITE_SYSTEM_PROMPT = """
你是一个数据库问答助手，负责把当前用户问题改写成“脱离上下文也能独立理解”的完整问题。

规则：
1. 必须结合历史对话理解代词、省略和追问。
2. 如果当前问题已经完整清晰，则保持原样。
3. 不要回答问题，只做改写。
4. 只输出改写后的问题，不要输出解释。
5. 改写结果应保持用户原意，不要额外扩展。
"""

SQL_SYSTEM_PROMPT = """
你是一个专业的 SQLite 数据库专家，需要把用户的自然语言问题转换为可执行的 SQLite SQL。

请严格遵守以下规则：

2. 只能生成 SELECT 或 WITH 开头的查询语句。
3. 必须结合给定的数据库 schema，表名和字段名必须真实存在。

5. 如果问题需要同时统计多个表，可以用子查询一次返回多个字段。
6. 当前数据库是 SQLite。
7. 优先生成简单、正确、可执行的 SQL。
"""

ANSWER_SYSTEM_PROMPT = """
你是一个数据库问答助手。
请根据用户问题、改写后的完整问题、SQL 和 SQL 查询结果，用简洁、自然、准确的中文回答。
不要编造结果，不要额外发挥。
"""

# 4. 多轮 SQL Agent
# =========================
class MultiTurnSQLAgent:
    def __init__(
            self,
            db_url: str,
            api_key: str,
            model: str = "glm-3-turbo",
            max_history_turns: int = 5
    ):
        self.db = DBParser(db_url)
        self.llm = GLMClient(api_key, model=model)
        self.schema_text = self.db.get_schema_text()
        self.max_history_turns = max_history_turns

        # 每一轮保存：
        # {
        #   "user_question": ...,
        #   "standalone_question": ...,
        #   "sql": ...,
        #   "answer": ...
        # }
        self.history: List[Dict[str, str]] = []

    def clear_history(self):
        """清空会话历史"""
        self.history = []

    def _extract_sql(self, raw_text: str) -> str:
        """
        从模型返回中提取 SQL
        """
        content = raw_text.strip()

        code_match = re.search(r"```sql\s*(.*?)```", content, flags=re.I | re.S)
        if code_match:
            content = code_match.group(1).strip()

        sql_match = re.search(r"(?is)\b(select|with)\b.*", content)
        if not sql_match:
            raise ValueError(f"模型没有返回有效 SQL：{raw_text}")

        sql = sql_match.group(0).strip()

        if ";" in sql:
            sql = sql.split(";")[0].strip()

        return sql

    def _history_to_text(self) -> str:
        """
        将最近几轮历史整理成文本，供模型理解上下文
        """
        if not self.history:
            return "无历史对话。"

        recent_history = self.history[-self.max_history_turns:]
        lines = []

        for idx, turn in enumerate(recent_history, start=1):
            lines.append(f"第{idx}轮用户问题：{turn['user_question']}")
            lines.append(f"第{idx}轮改写问题：{turn['standalone_question']}")
            lines.append(f"第{idx}轮SQL：{turn['sql']}")
            lines.append(f"第{idx}轮回答：{turn['answer']}")
            lines.append("")

        return "\n".join(lines).strip()

    def _rewrite_question(self, user_question: str) -> str:
        """
        将当前问题改写成独立完整的问题
        """
        history_text = self._history_to_text()

        prompt = f"""
历史对话如下：
{history_text}

当前用户问题：
{user_question}

请将当前问题改写为完整问题。
"""

        rewritten = self.llm.chat(
            messages=[
                {"role": "system", "content": QUESTION_REWRITE_SYSTEM_PROMPT},
                {"role": "user", "content": prompt}
            ],
            temperature=0.1
        ).strip()

        return rewritten

    def _generate_sql(self, standalone_question: str, error_msg: Optional[str] = None) -> str:
        """
        让大模型根据完整问题生成 SQL
        """
        history_text = self._history_to_text()

        prompt = f"""
数据库 schema 如下：
{self.schema_text}

最近历史对话如下：
{history_text}

当前需要回答的完整问题：
{standalone_question}
"""

        if error_msg:
            prompt += f"""

上一版 SQL 执行报错如下：
{error_msg}

请修正 SQL。
"""

        raw = self.llm.chat(
            messages=[
                {"role": "system", "content": SQL_SYSTEM_PROMPT},
                {"role": "user", "content": prompt}
            ],
            temperature=0.1
        )
        return self._extract_sql(raw)

    def _compress_rows(self, rows: List[Dict[str, Any]], max_rows: int = 10) -> List[Dict[str, Any]]:
        """
        返回给回答模型时做一个截断，避免结果过长
        """
        if len(rows) <= max_rows:
            return rows
        return rows[:max_rows]

    def _rewrite_answer(
            self,
            user_question: str,
            standalone_question: str,
            sql: str,
            rows: List[Dict[str, Any]]
    ) -> str:
        """
        将查询结果组织成自然语言
        """
        shown_rows = self._compress_rows(rows, max_rows=10)
        result_json = json.dumps(shown_rows, ensure_ascii=False)

        prompt = f"""
用户本轮原始问题：
{user_question}

改写后的完整问题：
{standalone_question}

执行的 SQL：
{sql}

SQL 查询结果（JSON）：
{result_json}

说明：
- 如果结果只有一个统计值，直接说出统计结果。
- 如果结果是多列统计值，分别说明。
- 如果结果为空，明确说明没有查到相关数据。
- 回答使用中文。
"""

        answer = self.llm.chat(
            messages=[
                {"role": "system", "content": ANSWER_SYSTEM_PROMPT},
                {"role": "user", "content": prompt}
            ],
            temperature=0.2
        )
        return answer.strip()

    def ask(self, user_question: str) -> Dict[str, Any]:
        """
        多轮问答主入口
        """
        standalone_question = self._rewrite_question(user_question)
        last_error = None

        for _ in range(2):  # 最多修正 1 次 SQL
            sql = self._generate_sql(standalone_question, error_msg=last_error)
            try:
                columns, rows = self.db.execute_sql(sql)
                answer = self._rewrite_answer(
                    user_question=user_question,
                    standalone_question=standalone_question,
                    sql=sql,
                    rows=rows
                )

                turn_record = {
                    "user_question": user_question,
                    "standalone_question": standalone_question,
                    "sql": sql,
                    "answer": answer
                }
                self.history.append(turn_record)

                return {
                    "user_question": user_question,
                    "standalone_question": standalone_question,
                    "sql": sql,
                    "columns": columns,
                    "rows": rows,
                    "answer": answer
                }

            except Exception as e:
                last_error = str(e)

        raise RuntimeError(f"SQL 生成两次仍失败，最后一次错误：{last_error}")
